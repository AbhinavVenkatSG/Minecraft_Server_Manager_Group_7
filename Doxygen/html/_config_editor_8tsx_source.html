7const FILE_TYPES = [
8 { id: "server-properties", name: "server.properties", fetchFn: (id) => api.getServerProperties(id), saveFn: (id, content) => api.updateServerProperties(id, content) },
9 { id: "whitelist", name: "whitelist.json", fetchFn: (id) => api.getWhitelist(id), saveFn: (id, content) => api.updateWhitelist(id, content) },
10 { id: "start-parameters", name: "start.sh", fetchFn: (id) => api.getStartParameters(id), saveFn: (id, content) => api.updateStartParameters(id, content) },
11];

16export default function ConfigEditor({ server, refreshKey }) {
17 const [activeFile, setActiveFile] = useState(FILE_TYPES[0].id);
18 const [files, setFiles] = useState({});
19 const [originalFiles, setOriginalFiles] = useState({});
20 const [modified, setModified] = useState(new Set());
21 const [loading, setLoading] = useState(false);
22 const [saving, setSaving] = useState(false);
23
24 useEffect(() => {
25 if (server?.id) {
26 fetchAllFiles();
27 } else {
28 setFiles({});
29 setOriginalFiles({});
30 setModified(new Set());
31 }
32 }, [server?.id, refreshKey]);
33
37 const fetchAllFiles = async () => {
38 setLoading(true);
39 const fetchedFiles = {};
40 const originalFetched = {};
41
42 try {
43 for (const fileType of FILE_TYPES) {
44 try {
45 const response = await fileType.fetchFn(server.id);
46 if (response.status === 200) {
47 const content = response.data?.content || "";
48 fetchedFiles[fileType.id] = content;
49 originalFetched[fileType.id] = content;
50 }
51 } catch (error) {
52 console.error(`Failed to fetch ${fileType.name}:`, error);
53 fetchedFiles[fileType.id] = "";
54 originalFetched[fileType.id] = "";
55 }
56 }
57 setFiles(fetchedFiles);
58 setOriginalFiles(originalFetched);
59 } finally {
60 setLoading(false);
61 }
62 };
63
64 const handleChange = (value) => {
65 setFiles((prev) => ({ ...prev, [activeFile]: value }));
66 setModified((prev) => new Set(prev).add(activeFile));
67 };
68
69 const handleSave = async () => {
70 if (!server?.id) return;
71 setSaving(true);
72 const fileType = FILE_TYPES.find(f => f.id === activeFile);
73
74 try {
75 await fileType.saveFn(server.id, files[activeFile]);
76 setOriginalFiles((prev) => ({ ...prev, [activeFile]: files[activeFile] }));
77 setModified((prev) => {
78 const next = new Set(prev);
79 next.delete(activeFile);
80 return next;
81 });
82 toast.success(`Saved ${fileType.name}`);
83 } catch (error) {
84 toast.error(`Failed to save ${fileType.name}: ${error.message}`);
85 } finally {
86 setSaving(false);
87 }
88 };
89
90 const handleRevert = () => {
91 setFiles((prev) => ({ ...prev, [activeFile]: originalFiles[activeFile] }));
92 setModified((prev) => {
93 const next = new Set(prev);
94 next.delete(activeFile);
95 return next;
96 });
97 toast.info(`Reverted changes`);
98 };
99
100 const currentContent = files[activeFile] || "";
101 const lineCount = currentContent.split("\n").length;
102
103 if (!server) {
104 return (
105 <div className="rounded-lg border border-border bg-card flex flex-col h-full">
106 <div className="px-4 py-3 border-b border-border">
107 <h3 className="text-sm font-semibold text-foreground">Config Editor</h3>
108 </div>
109 <div className="flex-1 flex items-center justify-center text-muted-foreground text-sm">
110 Select a server to edit config files
111 </div>
112 </div>
113 );
114 }
115
116 return (
117 <div className="rounded-lg border border-border bg-card flex flex-col h-full">
118 <Tabs value={activeFile} onValueChange={setActiveFile} className="flex flex-col h-full">
119 <div className="flex items-center justify-between border-b border-border px-2">
120 <TabsList className="bg-transparent h-auto p-0 gap-0">
121 {FILE_TYPES.map((fileType) => (
122 <TabsTrigger
123 key={fileType.id}
124 value={fileType.id}
125 className="rounded-none border-b-2 border-transparent data-[state=active]:border-primary data-[state=active]:bg-transparent data-[state=active]:shadow-none px-3 py-2.5 text-xs font-mono flex items-center gap-1.5"
126 >
127 <FileText size={12} />
128 {fileType.name}
129 {modified.has(fileType.id) && (
130 <span className="w-1.5 h-1.5 rounded-full bg-warning" />
131 )}
132 </TabsTrigger>
133 ))}
134 </TabsList>
135
136 <div className="flex items-center gap-1 pr-2">
137 <button
138 onClick={handleRevert}
139 disabled={!modified.has(activeFile)}
140 className="p-1.5 rounded text-muted-foreground hover:text-foreground disabled:opacity-30 transition-colors"
141 title="Revert changes"
142 >
143 <RotateCcw size={14} />
144 </button>
145 <button
146 onClick={handleSave}
147 disabled={!modified.has(activeFile) || saving}
148 className="p-1.5 rounded text-muted-foreground hover:text-primary disabled:opacity-30 transition-colors"
149 title="Save file"
150 >
151 {saving ? <Loader2 size={14} className="animate-spin" /> : <Save size={14} />}
152 </button>
153 </div>
154 </div>
155
156 {loading ? (
157 <div className="flex-1 flex items-center justify-center">
158 <Loader2 size={24} className="animate-spin text-muted-foreground" />
159 </div>
160 ) : (
161 FILE_TYPES.map((fileType) => (
162 <TabsContent key={fileType.id} value={fileType.id} className="flex-1 m-0 min-h-0">
163 <div className="flex h-full">
164 <div className="py-3 px-2 text-right select-none border-r border-border min-w-[3rem] overflow-hidden">
165 {Array.from({ length: lineCount }, (_, i) => (
166 <div key={i} className="text-[11px] leading-5 text-muted-foreground font-mono">
167 {i + 1}
168 </div>
169 ))}
170 </div>
171 <textarea
172 value={files[fileType.id] || ""}
173 onChange={(e) => handleChange(e.target.value)}
174 spellCheck={false}
175 className="flex-1 bg-transparent p-3 text-xs font-mono text-foreground leading-5 resize-none focus:outline-none min-h-0 overflow-auto"
176 placeholder={`No ${fileType.name} file content`}
177 />
178 </div>
179 </TabsContent>
180 ))
181 )}
182 </Tabs>
183 </div>
184 );
185}
