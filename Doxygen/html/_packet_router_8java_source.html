29 private static void handleCommand(WebSocket conn, Packet packet, CommandHandler commandHandler, BinaryWebSocketServer.ClientSession session) {
30 String payload = packet.getPayload();
31 if (payload == null || payload.isEmpty()) {
32 sendError(conn, "Command payload is empty.");
33 return;
34 }
35
36 String[] parts = payload.split(" ", 3);
37 if (parts.length == 0) {
38 sendError(conn, "Invalid command format.");
39 return;
40 }
41
42 String command = parts[0].toUpperCase();
43
44 try {
45 switch (command) {
46 case "CMD_START":
47 if (parts.length < 2) {
48 sendError(conn, "Usage: CMD_START <serverId>");
49 return;
50 }
51 long startServerId = parseServerId(parts[1]);
52 commandHandler.handleStartServer(conn, startServerId);
53 break;
54
55 case "CMD_STOP":
56 if (parts.length < 2) {
57 sendError(conn, "Usage: CMD_STOP <serverId>");
58 return;
59 }
60 long stopServerId = parseServerId(parts[1]);
61 commandHandler.handleStopServer(conn, stopServerId);
62 break;
63
64 case "CMD_STATUS":
65 if (parts.length < 2) {
66 sendError(conn, "Usage: CMD_STATUS <serverId>");
67 return;
68 }
69 long statusServerId = parseServerId(parts[1]);
70 commandHandler.handleStatus(conn, statusServerId);

73 case "CMD_BACKUP":
74 if (parts.length < 2) {
75 sendError(conn, "Usage: CMD_BACKUP <serverId>");
76 return;
77 }
78 long backupServerId = parseServerId(parts[1]);
79 commandHandler.handleBackup(conn, backupServerId);

81
82 case "CMD_CONSOLE":
83 if (parts.length < 3) {
84 sendError(conn, "Usage: CMD_CONSOLE <serverId> <command>");

103 sendError(conn, "Usage: CMD_SUBSCRIBE <serverId>");

107 commandHandler.handleSubscribe(conn, session, subscribeServerId);

116 commandHandler.handleUnsubscribe(conn, session, unsubscribeServerId);

121 sendError(conn, "Usage: CMD_LIST_PLAYERS <serverId>");

123 }
124 long playersServerId = parseServerId(parts[1]);
125 commandHandler.handleListPlayers(conn, playersServerId);
126 break;
127
128 default:
129 sendError(conn, "Unknown command: " + command + ". Supported: CMD_START, CMD_STOP, CMD_STATUS, CMD_BACKUP, CMD_CONSOLE, CMD_TELEMETRY, CMD_SUBSCRIBE, CMD_UNSUBSCRIBE, CMD_LIST_PLAYERS");
130 }
131 } catch (NumberFormatException e) {
132 sendError(conn, "Invalid server ID format: " + parts[1]);
133 } catch (Exception e) {
134 sendError(conn, "Command execution failed: " + e.getMessage());
