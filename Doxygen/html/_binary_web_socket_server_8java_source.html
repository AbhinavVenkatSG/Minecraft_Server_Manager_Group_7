79 @Override
80 public void onOpen(WebSocket conn, ClientHandshake handshake) {
81 String apiKey = handshake.getFieldValue("apiKey");
82 if (apiKey == null) {
83 String queryString = handshake.getResourceDescriptor();
84 if (queryString != null && queryString.contains("apiKey=")) {
85 int start = queryString.indexOf("apiKey=") + 7;
86 int end = queryString.indexOf("&", start);
87 if (end == -1) end = queryString.length();
88 apiKey = queryString.substring(start, end);
89 }
90 }
91
92 if (!EXPECTED_API_KEY.equals(apiKey)) {
93 sendPacket(conn, PacketBuilder.buildError("Invalid or missing API key."));
94 conn.close(1008, "Invalid API key");
95 return;
96 }
97
98 clients.put(conn, new ClientSession());
99 }

111 @Override
112 public void onMessage(WebSocket conn, ByteBuffer message) {
113 ClientSession session = clients.get(conn);
114 if (session == null) {
115 sendPacket(conn, PacketBuilder.buildError("Session not found."));
116 return;
117 }
118
119 try {
120 byte[] data = new byte[message.remaining()];
121 message.get(data);
122 Packet packet = Packet.fromBytes(data);
123
124 if (!packet.isValid()) {
125 sendPacket(conn, PacketBuilder.buildError("CRC validation failed."));
126 return;
127 }
128
129 PacketRouter.route(conn, packet, commandHandler, session);
130 } catch (IllegalArgumentException e) {
131 sendPacket(conn, PacketBuilder.buildError("Invalid packet format: " + e.getMessage()));
132 } catch (Exception e) {
133 sendPacket(conn, PacketBuilder.buildError("Processing error: " + e.getMessage()));
134 }
135 }
