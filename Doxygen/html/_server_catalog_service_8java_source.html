55 public Optional<ManagedServer> createServer(ServerRequest request, long ownerId) {
56 if (request == null || support.isBlank(request.getName()) || support.isBlank(request.getHost()) || request.getPort() <= 0) {
57 return Optional.empty();
58 }
59 if (!support.listServers().isEmpty()) {
60 return Optional.empty();
61 }
62
63 String minecraftDirectory = support.safe(request.getMinecraftDirectory());
64 if (!minecraftDirectory.isBlank()) {
65 support.validateMinecraftDirectory(Path.of(minecraftDirectory));
66 }
67
68 ManagedServer server = new ManagedServer(
69 0L,
70 request.getName().trim(),
71 request.getHost().trim(),
72 request.getPort(),
73 support.safe(request.getRconPassword()),
74 request.getRconPort() > 0 ? request.getRconPort() : 25575,
75 minecraftDirectory.isBlank() ? ServerStatus.STARTUP : ServerStatus.STOPPED,
76 support.defaultServerProperties(minecraftDirectory, request.getServerProperties()),
77 support.safe(request.getBackupPath()),
78 minecraftDirectory,
79 ownerId,
80 LocalDateTime.now(),
81 null
82 );
83
84 return Optional.of(support.saveServer(server));
85 }

90 public void ensureInteractiveSetup() {
91 if (!support.listServers().isEmpty()) {
92 return;
93 }
94
95 Console console = System.console();
96 Scanner scanner = console == null ? new Scanner(System.in) : null;
97
98 while (true) {
99 String rawDirectory = console != null
100 ? console.readLine("Enter the Minecraft server directory: ")
101 : promptAndRead(scanner, "Enter the Minecraft server directory: ");
102
103 if (support.isBlank(rawDirectory)) {
104 continue;
105 }
106
107 try {
108 Path directory = support.validateMinecraftDirectory(Path.of(rawDirectory.trim()));
109 String serverName = directory.getFileName() == null ? "Local Minecraft Server" : directory.getFileName().toString();
110 createServer(new ServerRequest(
111 serverName,
112 "localhost",
113 25565,
114 "",
115 25575,
116 support.readOptionalFile(directory.resolve("server.properties")),
117 "",
118 directory.toAbsolutePath().toString()
119 ), 1L);
120 System.out.println("Minecraft server directory configured: " + directory.toAbsolutePath());
121 return;
122 } catch (IllegalArgumentException exception) {
123 System.out.println(exception.getMessage());
124 }

145 public ManagedServer updateMinecraftDirectory(long serverId, String rawDirectory) {
146 ManagedServer server = support.requireServer(serverId);
147 if (server.getStatus() == ServerStatus.RUNNING) {
148 throw new IllegalStateException("Stop the server before changing the Minecraft directory.");
149 }
150
151 Path directory = support.validateMinecraftDirectory(Path.of(rawDirectory.trim()));
152 server.setMinecraftDirectory(directory.toAbsolutePath().toString());
153 server.setServerProperties(support.readOptionalFile(directory.resolve("server.properties")));
154 server.setStatus(ServerStatus.STOPPED);
155 return support.saveServer(server);
156 }
