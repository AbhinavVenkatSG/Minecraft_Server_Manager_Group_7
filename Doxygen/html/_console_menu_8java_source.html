68 public void run() {
69 boolean running = true;
70 while (running) {
71 ManagedServer server = requirePrimaryServer();
72 printMenu(server);
73 String choice = readLine("Select an option: ");
74
75 try {
76 switch (choice) {
77 case "1" -> startServer(server);
78 case "2" -> stopServer(server);
79 case "3" -> restartServer(server);
80 case "4" -> showPlayerCount(server);
81 case "5" -> showRamUsage(server);
82 case "6" -> showCpuUsage(server);
83 case "7" -> readWhitelist(server);
84 case "8" -> writeWhitelist(server);
85 case "9" -> readServerProperties(server);
86 case "10" -> writeServerProperties(server);
87 case "11" -> readStartParameters(server);
88 case "12" -> writeStartParameters(server);
89 case "13" -> createBackup(server);
90 case "14" -> showLatestConsoleLog(server);
91 case "15" -> changeMinecraftDirectory(server);
92 case "16" -> showTelemetryLog(server);
93 case "17" -> showServerSize(server);
94 case "18" -> addUser();
95 case "19" -> running = false;
96 default -> System.out.println("Invalid option.");
97 }
98 } catch (Exception exception) {
99 System.out.println("Operation failed: " + exception.getMessage());
100 }
101
102 if (running) {
103 System.out.println();
104 }
105 }

290 private String readMultiline(String prompt) {
291 System.out.println(prompt);
292 StringBuilder builder = new StringBuilder();
293
294 while (true) {
295 String line = readLine("");
296 if ("END".equals(line)) {
297 break;
298 }
299 if (builder.length() > 0) {
300 builder.append(System.lineSeparator());
301 }
302 builder.append(line);
303 }

316 private String formatBytes(long bytes) {
317 if (bytes <= 0L) {
318 return "0 B";
319 }
320 double value = bytes;
321 String[] units = {"B", "KB", "MB", "GB", "TB"};
322 int unitIndex = 0;
323 while (value >= 1024d && unitIndex < units.length - 1) {
324 value /= 1024d;
325 unitIndex++;
326 }
327 return String.format("%.2f %s", value, units[unitIndex]);
328 }
