77 public Optional<Backup> createBackup(long serverId) {
78 Optional<ManagedServer> server = serverStore.findById(serverId);
79 if (server.isEmpty()) {
80 return Optional.empty();
81 }
82
83 ManagedServer managedServer = server.get();
84 if (managedServer.getStatus() != ServerStatus.STOPPED) {
85 throw new IllegalStateException("Server must be stopped before creating a backup.");
86 }
87
88 Path minecraftDirectory = Path.of(managedServer.getMinecraftDirectory());
89 Path worldDirectory = minecraftDirectory.resolve("world");
90 if (!Files.isDirectory(worldDirectory)) {
91 throw new IllegalStateException("World directory not found at: " + worldDirectory);
92 }
93
94 try {
95 Path backupDirectory = resolveBackupDirectory(managedServer, minecraftDirectory);
96 Files.createDirectories(backupDirectory);
97
98 String filename = "backup-" + serverId + "-" + System.currentTimeMillis() + ".zip";
99 Path backupFile = backupDirectory.resolve(filename);
100 zipDirectory(worldDirectory, backupFile);
101
102 Backup backup = new Backup(
103 0L,
104 serverId,
105 filename,
106 backupFile.toAbsolutePath().toString(),
107 Files.size(backupFile),
108 LocalDateTime.now()
109 );
110
111 return Optional.of(backupStore.save(backup));
112 } catch (IOException exception) {
113 throw new IllegalStateException("Unable to create backup.", exception);
114 }
115 }

124 private void zipDirectory(Path sourceDirectory, Path targetZip) throws IOException {
125 try (OutputStream outputStream = Files.newOutputStream(
126 targetZip,
127 StandardOpenOption.CREATE,
128 StandardOpenOption.TRUNCATE_EXISTING,
129 StandardOpenOption.WRITE
130 ); ZipOutputStream zipOutputStream = new ZipOutputStream(outputStream)) {
131 Files.walk(sourceDirectory)
132 .filter(Files::isRegularFile)
133 .forEach(path -> writeZipEntry(sourceDirectory, path, zipOutputStream));
134 }

137 private void writeZipEntry(Path sourceDirectory, Path path, ZipOutputStream zipOutputStream) {
138 String entryName = sourceDirectory.relativize(path).toString().replace('\\', '/');
139 try (InputStream inputStream = Files.newInputStream(path)) {
140 zipOutputStream.putNextEntry(new ZipEntry(entryName));
141 inputStream.transferTo(zipOutputStream);
142 zipOutputStream.closeEntry();
143 } catch (IOException exception) {
144 throw new IllegalStateException("Unable to zip file: " + path, exception);
145 }
