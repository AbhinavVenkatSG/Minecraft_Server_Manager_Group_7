109 private static int resolveApiPort(String[] args) {
110 if (args.length > 0) {
111 return parsePort(args[0], 9000);
112 }
113
114 String envPort = System.getenv("APP_PORT");
115 if (envPort != null && !envPort.isBlank()) {
116 return parsePort(envPort, 9000);
117 }
118
119 return 9000;
120 }

122 private static int resolveWebSocketPort(String[] args, int apiPort) {

128 if (envPort != null && !envPort.isBlank()) {

130 }
131

133 }

143 private static Path resolveDataDirectory() throws Exception {
144 Path currentWorkingData = Path.of("data").toAbsolutePath().normalize();
145 if (hasPersistedData(currentWorkingData)) {
146 return currentWorkingData;
147 }
148
149 Path codeLocation = Path.of(Main.class.getProtectionDomain().getCodeSource().getLocation().toURI());
150 Path serverDirectory = Files.isDirectory(codeLocation) ? codeLocation.getParent() : codeLocation.getParent();
151 Path repositoryData = serverDirectory.getParent().resolve("data").toAbsolutePath().normalize();
152 if (hasPersistedData(repositoryData) || !Files.exists(currentWorkingData)) {
153 return repositoryData;
154 }
155
156 return currentWorkingData;
157 }
158
159 private static boolean hasPersistedData(Path dataDirectory) {
160 return Files.exists(dataDirectory.resolve("servers").resolve("1.properties"))
