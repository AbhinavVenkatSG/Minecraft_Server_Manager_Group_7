101 private TelemetrySnapshot collectTelemetry(ManagedServer server) {
102 long directorySize = 0L;
103 long driveTotal = 0L;
104 long driveUsable = 0L;
105 long processMemory = 0L;
106 Double processCpuLoad = null;
107 String jvmAllocatedRam = "";
108 String jvmInitialRam = "";
109 List<String> newLogLines = List.of();
110 List<String> onlinePlayers = List.of();
111
112 if (!support.isBlank(server.getMinecraftDirectory())) {
113 Path minecraftDirectory = support.requireMinecraftDirectory(server);
114 directorySize = support.computeDirectorySize(minecraftDirectory);
115 long[] driveMetrics = support.getDriveMetrics(minecraftDirectory);
116 driveTotal = driveMetrics[0];
117 driveUsable = driveMetrics[1];
118 String startParameters = "";
119 try {
120 startParameters = support.readRequiredFile(support.resolveStartScript(minecraftDirectory));
121 } catch (IllegalStateException ignored) {
122 startParameters = "";
123 }
124 jvmAllocatedRam = extractJvmArgument(startParameters, XMX_PATTERN);
125 jvmInitialRam = extractJvmArgument(startParameters, XMS_PATTERN);
126 newLogLines = support.readNewLogLines(server.getId(), minecraftDirectory.resolve("logs").resolve("latest.log"));
127 onlinePlayers = support.readOnlinePlayers(minecraftDirectory.resolve("logs").resolve("latest.log"));
128 }
129
130 ServerRuntimeState.RuntimeContext context = support.getRuntimeState().getRuntimeContexts().get(server.getId());
131 if (context != null && context.getProcess().isAlive()) {
132 OptionalDouble processCpuReading = support.readProcessCpuLoad(context);
133 processCpuLoad = processCpuReading.isPresent() ? processCpuReading.getAsDouble() : null;
134 processMemory = support.readProcessMemoryBytes(context);
135 }
136
137 OperatingSystemMXBean osBean = (OperatingSystemMXBean) ManagementFactory.getOperatingSystemMXBean();
138 long totalMemory = osBean.getTotalMemorySize();
139 long freeMemory = osBean.getFreeMemorySize();
140 Double systemCpuLoad = null;
141 double rawSystemCpuLoad = osBean.getCpuLoad();
142 if (rawSystemCpuLoad >= 0.0d && Double.isFinite(rawSystemCpuLoad)) {
143 systemCpuLoad = rawSystemCpuLoad * 100.0d;
144 }
145
146 String payload = buildTelemetryPayload(
147 server,
148 systemCpuLoad,
149 totalMemory - freeMemory,
150 totalMemory,
151 processMemory,
152 processCpuLoad,
153 jvmAllocatedRam,
154 jvmInitialRam,
155 directorySize,
156 driveTotal,
157 driveUsable,
158 onlinePlayers,
159 newLogLines
160 );
161
162 String packetBase64 = Base64.getEncoder().encodeToString(PacketBuilder.buildTelemetry(payload).toBytes());
163 TelemetrySnapshot snapshot = new TelemetrySnapshot(
164 server.getId(),
165 server.getStatus(),
166 systemCpuLoad,
167 totalMemory - freeMemory,
168 totalMemory,
169 processMemory,
170 processCpuLoad,
171 jvmAllocatedRam,
172 jvmInitialRam,
173 directorySize,
174 driveTotal,
175 driveUsable,
176 onlinePlayers.size(),
177 onlinePlayers,
178 newLogLines,
179 packetBase64,
180 LocalDateTime.now()
181 );
182 appendTelemetryHistory(server.getId(), snapshot);
183 return snapshot;
184 }

223 private void appendTelemetryHistory(long serverId, TelemetrySnapshot snapshot) {
224 Deque<String> history = support.getRuntimeState().getTelemetryHistory().computeIfAbsent(serverId, unused -> new ArrayDeque<>());
225 String entry = snapshot.getCapturedAt()
226 + " | state=" + snapshot.getOperationalState().name()
227 + " | cpu=" + formatCpuLoad(snapshot.getMinecraftProcessCpuLoadPercent())
228 + " | ram=" + snapshot.getMinecraftProcessMemoryBytes() + " bytes"
229 + " | players=" + snapshot.getPlayerCount()
230 + " | size=" + snapshot.getMinecraftDirectorySizeBytes() + " bytes"
231 + " | logs=" + snapshot.getNewLogLines().size();
232
233 synchronized (history) {
234 history.addLast(entry);
235 while (history.size() > MAX_TELEMETRY_HISTORY) {
236 history.removeFirst();
237 }
238 }
239 }
