116 private final class ApiHandler implements HttpHandler {
117 @Override
118 public void handle(HttpExchange exchange) throws IOException {
119 addCorsHeaders(exchange);
120
121 if ("OPTIONS".equalsIgnoreCase(exchange.getRequestMethod())) {
122 send(exchange, 204, "");
123 return;
124 }
125
126 String method = exchange.getRequestMethod().toUpperCase();
127 String path = exchange.getRequestURI().getPath();
128 String body = readBody(exchange);
129 String[] parts = path.substring(1).split("/");
130
131 try {
132 if (parts.length >= 3 && "api".equals(parts[0]) && "auth".equals(parts[1])) {
133 handleAuth(exchange, method, parts, body);
134 return;
135 }
136
137 if (parts.length >= 2 && "api".equals(parts[0]) && "servers".equals(parts[1])) {
138 handleServers(exchange, method, parts, body);
139 return;
140 }
141
142 if (parts.length >= 2 && "api".equals(parts[0]) && "backups".equals(parts[1])) {
143 handleBackupDownloads(exchange, method, parts);
144 return;
145 }
146
147 send(exchange, 404, Responses.message("Route not found."));
148 } catch (IllegalArgumentException | IllegalStateException exception) {
149 send(exchange, 400, Responses.message(exception.getMessage()));
150 } catch (Exception exception) {
151 send(exchange, 500, Responses.message("Server error: " + exception.getMessage()));
152 }
153 }

156 private void handleAuth(HttpExchange exchange, String method, String[] parts, String body) throws IOException {
157 if ("POST".equals(method) && parts.length == 3 && "login".equals(parts[2])) {
158 Optional<LoginResponse> response = authService.login(new LoginRequest(
159 Json.readString(body, "username"),
160 Json.readString(body, "password")
161 ));
162 if (response.isPresent()) {
163 send(exchange, 200, Responses.login(response.get()));
164 } else {
165 send(exchange, 401, Responses.message("Invalid username or password."));
166 }
167 return;
168 }
169
170 if ("POST".equals(method) && parts.length == 3 && "register".equals(parts[2])) {
171 Optional<LoginResponse> response = authService.register(new RegisterRequest(
172 Json.readString(body, "username"),
173 Json.readString(body, "password")
174 ));
175 if (response.isPresent()) {
176 send(exchange, 200, Responses.login(response.get()));
177 } else {
178 send(exchange, 400, Responses.message("Could not register user."));
179 }
180 return;
181 }
182
183 send(exchange, 404, Responses.message("Auth route not found."));
184 }

331 private void handleBackupDownloads(HttpExchange exchange, String method, String[] parts) throws IOException {
332 if (!"GET".equals(method) || parts.length != 4 || !"download".equals(parts[3])) {
333 send(exchange, 404, Responses.message("Backup route not found."));
334 return;
335 }
336
337 long backupId = parseId(parts[2]);
338 if (backupId < 0) {
339 send(exchange, 400, Responses.message("Invalid backup id."));
340 return;
341 }
342
343 Optional<Backup> backup = backupService.getBackup(backupId);
344 Optional<Path> backupPath = backupService.resolveBackupFile(backupId);
345 if (backup.isEmpty() || backupPath.isEmpty()) {
346 send(exchange, 404, Responses.message("Backup not found."));
347 return;
348 }
349
350 byte[] fileBytes = Files.readAllBytes(backupPath.get());
351 exchange.getResponseHeaders().set("Content-Type", "application/zip");
352 exchange.getResponseHeaders().set(
353 "Content-Disposition",
354 "attachment; filename=\"" + backup.get().getFilename() + "\""
355 );
356 exchange.sendResponseHeaders(200, fileBytes.length);
357 try (OutputStream outputStream = exchange.getResponseBody()) {
358 outputStream.write(fileBytes);
359 }
360 }
