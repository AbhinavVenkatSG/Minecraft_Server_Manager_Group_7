102 Backup backup = new Backup(
103 Long.parseLong(properties.getProperty("id", "0")),
104 Long.parseLong(properties.getProperty("serverId", "0")),
105 properties.getProperty("filename", ""),
106 properties.getProperty("filePath", ""),
107 Long.parseLong(properties.getProperty("fileSize", "0")),
108 LocalDateTime.parse(properties.getProperty("createdAt"))
109 );
110
111 backups.put(backup.getId(), backup);
112 nextId.set(Math.max(nextId.get(), backup.getId() + 1));

118 private void writeBackup(Backup backup) {
119 try {
120 Properties properties = new Properties();
121 properties.setProperty("id", Long.toString(backup.getId()));
122 properties.setProperty("serverId", Long.toString(backup.getServerId()));
123 properties.setProperty("filename", backup.getFilename());
124 properties.setProperty("filePath", backup.getFilePath());
125 properties.setProperty("fileSize", Long.toString(backup.getFileSize()));
126 properties.setProperty("createdAt", backup.getCreatedAt().toString());
127
128 StringWriter writer = new StringWriter();
129 properties.store(writer, null);
130 Files.writeString(backupDirectory.resolve(backup.getId() + ".properties"), writer.toString());
131 } catch (Exception exception) {
132 throw new IllegalStateException("Unable to persist backup.", exception);
