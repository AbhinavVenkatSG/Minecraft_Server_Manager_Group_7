32 public Optional<ManagedServer> startServer(long serverId) {
33 Optional<ManagedServer> server = support.getServer(serverId);
34 if (server.isEmpty()) {
35 return Optional.empty();
36 }
37
38 ManagedServer managedServer = server.get();
39 if (managedServer.getStatus() == ServerStatus.RUNNING || support.getRuntimeState().getRuntimeContexts().containsKey(serverId)) {
40 throw new IllegalStateException("Server is already running.");
41 }
42
43 Path minecraftDirectory = support.requireMinecraftDirectory(managedServer);
44 Path startScript = support.resolveStartScript(minecraftDirectory);
45
46 try {
47 managedServer.setStatus(ServerStatus.BLOCKED);
48 support.saveServer(managedServer);
49
50 // The local setup uses batch files, so the process must be launched through cmd.
51 Process process = new ProcessBuilder("cmd.exe", "/c", startScript.getFileName().toString())
52 .directory(minecraftDirectory.toFile())
53 .redirectErrorStream(true)
54 .start();
55
56 ServerRuntimeState.RuntimeContext context = new ServerRuntimeState.RuntimeContext(process);
57 support.getRuntimeState().getRuntimeContexts().put(serverId, context);
58 support.startOutputPump(serverId, managedServer, context);
59
60 managedServer.setStatus(ServerStatus.RUNNING);
61 managedServer.setLastStarted(LocalDateTime.now());
62 support.saveServer(managedServer);
63 return Optional.of(managedServer);
64 } catch (IOException exception) {
65 support.getRuntimeState().getRuntimeContexts().remove(serverId);
66 managedServer.setStatus(ServerStatus.STOPPED);
67 support.saveServer(managedServer);
68 throw new IllegalStateException("Unable to start Minecraft server.", exception);
69 }
70 }
