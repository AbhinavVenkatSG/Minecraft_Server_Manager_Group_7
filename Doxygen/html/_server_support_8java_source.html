292 public List<String> readNewLogLines(long serverId, Path latestLogPath) {
293 if (!Files.exists(latestLogPath)) {
294 return List.of();
295 }
296
297 try {
298 long fileSize = Files.size(latestLogPath);
299 long offset = runtimeState.getLatestLogOffsets().getOrDefault(serverId, 0L);
300 if (offset > fileSize) {
301 offset = 0L;
302 }
303
304 List<String> allLines = Files.readAllLines(latestLogPath, StandardCharsets.UTF_8);
305 int startIndex = estimateLineIndex(allLines, offset);
306 List<String> newLines = allLines.subList(startIndex, allLines.size());
307 runtimeState.getLatestLogOffsets().put(serverId, fileSize);
308 return List.copyOf(newLines);
309 } catch (IOException exception) {
310 throw new IllegalStateException("Unable to read latest.log.", exception);

320 public List<String> readOnlinePlayers(Path latestLogPath) {
321 if (!Files.exists(latestLogPath)) {
322 return List.of();
323 }
324
325 try {
326 List<String> lines = Files.readAllLines(latestLogPath, StandardCharsets.UTF_8);
327 Set<String> players = new LinkedHashSet<>();
328 for (String line : lines) {
329 Matcher joinMatcher = PLAYER_JOIN_PATTERN.matcher(line);
330 if (joinMatcher.find()) {
331 players.add(joinMatcher.group(1));
332 }
333
334 Matcher leaveMatcher = PLAYER_LEAVE_PATTERN.matcher(line);
335 if (leaveMatcher.find()) {
336 players.remove(leaveMatcher.group(1));
337 }
338 }
339 return List.copyOf(players);
340 } catch (IOException exception) {
341 throw new IllegalStateException("Unable to inspect latest.log for players.", exception);

351 public Optional<ProcessHandle> resolveTelemetryProcess(ServerRuntimeState.RuntimeContext context) {
352 if (context == null || context.getProcess() == null) {
353 return Optional.empty();
354 }
355
356 ProcessHandle root = context.getProcess().toHandle();
357 if (!root.isAlive()) {
358 return Optional.empty();
359 }
360
361 List<ProcessHandle> liveDescendants = root.descendants()
362 .filter(ProcessHandle::isAlive)
363 .sorted(Comparator.comparingLong(this::telemetryProcessPriority).reversed())
364 .toList();
365 if (!liveDescendants.isEmpty()) {
366 return Optional.of(liveDescendants.get(0));
367 }
368
369 return Optional.of(root);
370 }

378 public OptionalDouble readProcessCpuLoad(ServerRuntimeState.RuntimeContext context) {
379 if (context.getProcess() == null) {
380 return OptionalDouble.empty();
381 }
382
383 Optional<ProcessHandle> telemetryProcess = resolveTelemetryProcess(context);
384 if (telemetryProcess.isEmpty()) {
385 return OptionalDouble.empty();
386 }
387
388 ProcessHandle processHandle = telemetryProcess.get();
389 Optional<Duration> cpuDuration = processHandle.info().totalCpuDuration();
390 if (cpuDuration.isEmpty()) {
391 return OptionalDouble.empty();
392 }
393
394 // ProcessHandle only gives cumulative CPU time, so usage has to be derived from deltas.
395 long now = System.nanoTime();
396 long pid = processHandle.pid();
397 long cpuNanos = cpuDuration.get().toNanos();
398 if (context.getLastCpuSampleNanos() == 0L || context.getLastCpuSampledPid() != pid) {
399 context.setLastCpuSampleNanos(now);
400 context.setLastCpuDurationNanos(cpuNanos);
401 context.setLastCpuSampledPid(pid);
402 return OptionalDouble.empty();
403 }
404
405 long elapsedWallNanos = now - context.getLastCpuSampleNanos();
406 long elapsedCpuNanos = cpuNanos - context.getLastCpuDurationNanos();
407 context.setLastCpuSampleNanos(now);
408 context.setLastCpuDurationNanos(cpuNanos);
409 context.setLastCpuSampledPid(pid);
410
411 if (elapsedWallNanos <= 0L || elapsedCpuNanos < 0L) {
412 return OptionalDouble.empty();
413 }
414
415 double usage = (elapsedCpuNanos * 100.0d) / (elapsedWallNanos * Runtime.getRuntime().availableProcessors());
416 if (Double.isNaN(usage) || Double.isInfinite(usage)) {
417 return OptionalDouble.empty();
418 }
419
420 return OptionalDouble.of(Math.max(0.0d, usage));
421 }

444 public long readProcessMemoryBytes(long pid) {
445 try {
446 Process probe = new ProcessBuilder(
447 "tasklist",
448 "/FI",
449 "PID eq " + pid,
450 "/FO",
451 "CSV",
452 "/NH"
453 ).redirectErrorStream(true).start();
454
455 try (BufferedReader reader = new BufferedReader(new InputStreamReader(probe.getInputStream(), StandardCharsets.UTF_8))) {
456 String line = reader.readLine();
457 if (line == null || line.startsWith("INFO:")) {
458 return 0L;
459 }
460
461 String[] parts = parseCsvLine(line);
462 if (parts.length < 5) {
463 return 0L;
464 }
465
466 String mem = parts[4].replace("\"", "").replace(",", "").replace(" K", "").trim();
467 return Long.parseLong(mem) * 1024L;
468 }
469 } catch (Exception exception) {
470 return 0L;
471 }

480 public String[] parseCsvLine(String line) {
481 List<String> parts = new ArrayList<>();
482 StringBuilder current = new StringBuilder();
483 boolean inQuotes = false;
484
485 for (int index = 0; index < line.length(); index++) {
486 char currentChar = line.charAt(index);
487 if (currentChar == '"') {
488 inQuotes = !inQuotes;
489 current.append(currentChar);
490 } else if (currentChar == ',' && !inQuotes) {
491 parts.add(current.toString());
492 current.setLength(0);
493 } else {
494 current.append(currentChar);
495 }
496 }
497 parts.add(current.toString());
498 return parts.toArray(String[]::new);
499 }

526 public void startOutputPump(long serverId, ManagedServer server, ServerRuntimeState.RuntimeContext context) {
527 Thread thread = new Thread(() -> {
528 try (BufferedReader reader = new BufferedReader(new InputStreamReader(
529 context.getProcess().getInputStream(),
530 StandardCharsets.UTF_8
531 ))) {
532 String line;
533 while ((line = reader.readLine()) != null) {
534 synchronized (context.getConsoleLines()) {
535 context.getConsoleLines().addLast(line);
536 while (context.getConsoleLines().size() > MAX_CONSOLE_LINES) {
537 context.getConsoleLines().removeFirst();
538 }
539 }
540 }
541 } catch (IOException ignored) {
542 // Process shutdown closes the stream.
543 } finally {
544 if (!context.isManuallyStopping()) {
545 finalizeStoppedServer(serverId, server);
546 }
547 }
548 }, "mc-console-" + serverId);
549 thread.setDaemon(true);
550 context.setOutputThread(thread);
551 thread.start();
552 }

604 private long telemetryProcessPriority(ProcessHandle processHandle) {

606 String command = processHandle.info().command().orElse("").toLowerCase();
607 if (command.endsWith("java.exe") || command.endsWith("javaw.exe") || command.endsWith("/java") || command.endsWith("/javaw")) {
