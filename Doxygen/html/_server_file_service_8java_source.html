105 public ManagedServer writeStartParameters(long serverId, String newParameters) {
106 ManagedServer server = support.requireServer(serverId);
107 support.ensureServerStopped(server);
108 Path startScript = support.resolveStartScript(support.requireMinecraftDirectory(server));
109 String script = support.readRequiredFile(startScript);
110 Matcher matcher = JAVA_ARGUMENT_PATTERN.matcher(script);
111 if (!matcher.find()) {
112 throw new IllegalStateException("Unable to find Java launch arguments in the start script.");
113 }
114
115 // Keep the launcher command and jar target intact while swapping only the tunable JVM flags.
116 String replacement = matcher.group(1) + newParameters.trim() + matcher.group(3);
117 String updatedScript = matcher.replaceFirst(Matcher.quoteReplacement(replacement));
118 support.writeTextFile(startScript, updatedScript);
119 return support.saveServer(server);
